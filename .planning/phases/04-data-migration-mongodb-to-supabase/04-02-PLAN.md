# Plan 04-02: User Migration + Data Load into Supabase

## Goal

Run two sequential operations against the production Supabase project: (1) import every MongoDB user via the Supabase Admin API with their original bcrypt hash so no password reset is required, then (2) execute all SQL files generated by plan 04-01 against the Supabase PostgreSQL superuser connection in FK order.

## Context

Plan 04-01 produced `migration/output/02_profiles.sql` through `migration/output/21_audit_logs.sql` and `migration/id_map.json`. Users must be loaded into `auth.users` first because every Supabase table has `owner_id UUID REFERENCES auth.users(id)`. The Supabase Admin API `create_user` with `password_hash` is the only correct way to import users — direct SQL INSERT into `auth.users` is a fallback if the API has a bug with hash saving (documented as GitHub issue #1678; test with one user before bulk import). All psycopg2 data loads use the `postgres` superuser DSN (port 5432, not the pooler) to bypass RLS entirely. The `migration/` scripts directory created in plan 04-01 is the working directory throughout.

## Tasks

### Task 1 — Write migration/load_users.py

**File(s):**
- `migration/load_users.py`

**Action:** Create

**Description:**

Write the user import script. It reads from MongoDB, calls the Supabase Admin API once per user with a 100ms delay and exponential backoff on 429, then writes `migration/user_id_map.json` (a subset of `id_map.json` scoped to users only, for easy lookup during debugging).

After each successful `create_user` call, it immediately calls `supabase.auth.admin.get_user(uid)` and asserts `encrypted_password` is non-empty (the hash-saving verification for issue #1678). If the hash is missing, it falls back to a direct SQL INSERT into `auth.users` + `auth.identities`.

```python
#!/usr/bin/env python3
"""
load_users.py — Import MongoDB users into Supabase Auth.

Run BEFORE any SQL files. This must complete first because auth.users is
referenced by every public.* table.

Usage:
  cd migration/
  python load_users.py [--test-one]

  --test-one: Import only the first user and verify login before bulk import.
"""

import os
import sys
import json
import time
import psycopg2
from datetime import datetime, timezone
from dotenv import load_dotenv
from supabase import create_client, Client

load_dotenv(".env.migration")
from migration.extract import extract_all
from migration.utils import build_id_map, normalize_bcrypt_prefix, to_ts

TEST_ONE = "--test-one" in sys.argv


def get_supabase_client() -> Client:
    return create_client(
        os.environ["SUPABASE_URL"],
        os.environ["SUPABASE_SERVICE_KEY"],
    )


def import_user_via_api(supabase: Client, mongo_user: dict, new_uuid: str) -> str:
    """
    Import one user via Admin API. Returns the Supabase auth.users UUID.
    Raises RuntimeError if user creation fails.
    """
    raw_hash = mongo_user.get("password", "")
    password_hash = normalize_bcrypt_prefix(raw_hash)

    result = supabase.auth.admin.create_user({
        "id": new_uuid,
        "email": mongo_user["email"],
        "password_hash": password_hash,
        "email_confirm": True,      # all existing users are treated as verified
        "user_metadata": {
            "first_name": mongo_user.get("first_name", ""),
            "last_name": mongo_user.get("last_name", ""),
            "role": mongo_user.get("role", "user"),
        },
    })

    if result.user is None:
        raise RuntimeError(
            f"create_user returned None for {mongo_user['email']}: {result}"
        )
    return result.user.id


def verify_hash_saved(supabase: Client, uid: str, email: str):
    """
    Verify the bcrypt hash was saved (checks for GitHub issue #1678).
    Uses a direct DB query because the Admin API get_user does not expose
    encrypted_password.
    """
    conn = psycopg2.connect(os.environ["SUPABASE_DB_DSN"])
    try:
        with conn.cursor() as cur:
            cur.execute(
                "SELECT encrypted_password FROM auth.users WHERE id = %s", (uid,)
            )
            row = cur.fetchone()
            if not row or not row[0]:
                return False
            return True
    finally:
        conn.close()


def import_user_via_sql_fallback(conn, mongo_user: dict, new_uuid: str):
    """
    Fallback: direct SQL INSERT when Admin API does not save the hash.
    Includes all required GoTrue fields to avoid silent login failures.
    """
    password_hash = normalize_bcrypt_prefix(mongo_user.get("password", ""))
    created_at = to_ts(mongo_user.get("created_at")) or datetime.now(timezone.utc).isoformat()
    updated_at = to_ts(mongo_user.get("updated_at")) or created_at

    with conn.cursor() as cur:
        cur.execute("""
            INSERT INTO auth.users (
                id, instance_id, aud, role, email,
                encrypted_password, email_confirmed_at,
                raw_app_meta_data, raw_user_meta_data,
                created_at, updated_at, last_sign_in_at,
                confirmation_token, recovery_token,
                email_change_token_new, email_change
            ) VALUES (
                %s,
                '00000000-0000-0000-0000-000000000000',
                'authenticated', 'authenticated', %s,
                %s, NOW(),
                '{"provider":"email","providers":["email"]}', '{}',
                %s, %s, %s,
                '', '', '', ''
            ) ON CONFLICT (id) DO NOTHING
        """, (
            new_uuid,
            mongo_user["email"],
            password_hash,
            created_at, updated_at, created_at,
        ))

        # auth.identities is required for email provider login
        cur.execute("""
            INSERT INTO auth.identities (
                id, user_id, identity_data, provider,
                last_sign_in_at, created_at, updated_at
            ) VALUES (
                %s, %s,
                jsonb_build_object('sub', %s::text, 'email', %s),
                'email',
                NOW(), %s, NOW()
            ) ON CONFLICT (id) DO NOTHING
        """, (
            new_uuid, new_uuid,
            new_uuid, mongo_user["email"],
            created_at,
        ))
    conn.commit()
    print(f"    [FALLBACK-SQL] {mongo_user['email']} inserted via direct SQL")


def import_users_with_backoff(supabase: Client, users: list, id_map: dict) -> dict:
    """
    Import all users. Returns {mongo_oid_str: supabase_uuid} map.
    Uses 100ms delay between calls (10/sec), exponential backoff on 429.
    """
    results = {}
    fallback_conn = None

    for i, user in enumerate(users):
        oid_str = str(user["_id"])
        new_uuid = id_map[oid_str]
        email = user.get("email", f"unknown-{oid_str}@example.com")

        for attempt in range(3):
            try:
                import_user_via_api(supabase, user, new_uuid)
                time.sleep(0.1)  # 100ms — 10 users/sec, conservative rate limit

                # Verify hash was saved (issue #1678 guard)
                if not verify_hash_saved(supabase, new_uuid, email):
                    print(f"    [WARN] Hash not saved for {email} — using SQL fallback")
                    if fallback_conn is None:
                        fallback_conn = psycopg2.connect(os.environ["SUPABASE_DB_DSN"])
                    # Delete the hash-less row first
                    supabase.auth.admin.delete_user(new_uuid)
                    import_user_via_sql_fallback(fallback_conn, user, new_uuid)

                results[oid_str] = new_uuid
                print(f"  [{i+1}/{len(users)}] OK  {email}")
                break

            except Exception as e:
                if "429" in str(e):
                    wait = 2 ** attempt
                    print(f"    [RATE LIMIT] Waiting {wait}s before retry...")
                    time.sleep(wait)
                elif "already registered" in str(e).lower() or "already exists" in str(e).lower():
                    # Idempotent: user already imported (re-run safety)
                    results[oid_str] = new_uuid
                    print(f"  [{i+1}/{len(users)}] SKIP {email} (already exists)")
                    break
                else:
                    print(f"  [ERROR] {email}: {e}")
                    raise

    if fallback_conn:
        fallback_conn.close()

    return results


def main():
    print("=== load_users.py ===")
    print("Extracting users from MongoDB...")
    data = extract_all()
    users = data["users"]
    print(f"  Found {len(users)} users")

    from migration.utils import build_id_map
    id_map = {}
    from migration.extract import COLLECTIONS
    for col in COLLECTIONS:
        id_map.update(build_id_map(data[col]))

    if TEST_ONE:
        users = users[:1]
        print(f"  --test-one: importing only 1 user for hash verification test")

    supabase = get_supabase_client()

    print(f"\nImporting {len(users)} user(s) into Supabase Auth...")
    user_id_map = import_users_with_backoff(supabase, users, id_map)

    with open("migration/user_id_map.json", "w") as f:
        json.dump(user_id_map, f, indent=2)
    print(f"\n  user_id_map.json written ({len(user_id_map)} entries)")

    if TEST_ONE:
        print("\n  --test-one complete.")
        print("  NEXT: manually verify login on the Supabase project with the test user's original password.")
        print("  If login succeeds, run: python load_users.py (full import)")
    else:
        print(f"\n  All {len(user_id_map)} users imported.")
        print("  NEXT: run load_data.py")


if __name__ == "__main__":
    main()
```

**Acceptance:**
- `python migration/load_users.py --test-one` runs without crash.
- After `--test-one`, `migration/user_id_map.json` exists with 1 entry.
- Script contains the 100ms `time.sleep(0.1)` delay between API calls.
- Script contains the `verify_hash_saved` function and SQL fallback path.
- Script handles `already registered` / `already exists` errors idempotently.

**Action:** Create

**Description:**

Write the data loader that connects to Supabase PostgreSQL via the `postgres` superuser DSN and executes all SQL files from `migration/output/` in numeric filename order. This must run AFTER `load_users.py` completes.

```python
#!/usr/bin/env python3
"""
load_data.py — Execute all SQL output files into Supabase PostgreSQL.

Connects via postgres superuser DSN (port 5432) which has BYPASSRLS privilege.
Runs files in numeric filename order (FK dependency order).

Prerequisites:
  1. load_users.py must have completed successfully
  2. migration/output/*.sql files must exist (from migrate.py)

Usage:
  cd migration/
  python load_data.py
"""

import os
import glob
import psycopg2
from dotenv import load_dotenv

load_dotenv(".env.migration")

OUTPUT_DIR = "migration/output"


def get_conn():
    """
    Direct postgres superuser connection — bypasses RLS.
    Use port 5432 (direct connection), NOT port 6543 (pooler).

    If connection fails with IPv6 error, set SUPABASE_DB_DSN to the
    Session Pooler connection string (different host, still port 5432).
    See: Supabase Dashboard > Settings > Database > Session pooler.
    """
    return psycopg2.connect(os.environ["SUPABASE_DB_DSN"])


def execute_sql_file(conn, filepath: str):
    """Read and execute one .sql file in a single transaction."""
    with open(filepath) as f:
        sql = f.read()

    filename = os.path.basename(filepath)
    try:
        with conn.cursor() as cur:
            cur.execute(sql)
        conn.commit()
        print(f"  [OK] {filename}")
    except Exception as e:
        conn.rollback()
        print(f"  [FAIL] {filename}: {e}")
        raise


def main():
    print("=== load_data.py ===")

    sql_files = sorted(glob.glob(os.path.join(OUTPUT_DIR, "*.sql")))
    if not sql_files:
        print(f"ERROR: No .sql files found in {OUTPUT_DIR}/")
        print("Run migrate.py first.")
        raise SystemExit(1)

    print(f"Found {len(sql_files)} SQL files to execute (in order):")
    for f in sql_files:
        print(f"  {os.path.basename(f)}")

    print(f"\nConnecting to Supabase PostgreSQL...")
    conn = get_conn()
    print("  Connected.")

    print(f"\nExecuting SQL files...")
    for filepath in sql_files:
        execute_sql_file(conn, filepath)

    conn.close()
    print("\nDone. All SQL files loaded.")
    print("NEXT: run validate.py")


if __name__ == "__main__":
    main()
```
