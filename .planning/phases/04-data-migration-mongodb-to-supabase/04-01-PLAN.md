# Plan 04-01: Python ETL Script — Extract, Transform, Output SQL

## Goal

Write `migration/migrate.py` (and supporting modules) that reads all 19 MongoDB collections, transforms every document to a relational row with correct UUID keys and type coercions, and writes ordered `.sql` INSERT files ready for bulk loading.

## Context

The MongoDB Atlas database `erp_radiance` holds 19 collections matching the 20 Supabase tables (auth.users is handled separately in plan 04-02). Every MongoDB `_id` ObjectId must become a deterministic UUID5 using a fixed project namespace so the script is idempotent. FK references (owner_id, company_id, contact_id, etc.) must be resolved through the same UUID lookup table. Generated columns (`search_vector`) must be excluded from all INSERT lists. The `accounts` table is self-referencing and requires a two-pass approach: insert all rows first with `parent_id = NULL`, then UPDATE parent_id values. The `payslips` table has a unique constraint on `(employee_id, period_month, period_year)` — duplicates from MongoDB must be deduplicated in Python before output.

`migration/utils.py` — UUID5 helpers and type coercions:

```python
import uuid
import json
from datetime import datetime, timezone

# Fixed namespace — NEVER change after first run
RADIANCE_NAMESPACE = uuid.UUID("a1b2c3d4-e5f6-7890-abcd-ef1234567890")


def oid_to_uuid(oid) -> str:
    """Convert MongoDB ObjectId or string to deterministic UUID5."""
    return str(uuid.uuid5(RADIANCE_NAMESPACE, str(oid)))


def build_id_map(docs: list) -> dict:
    """Returns {objectid_str: uuid_str} for every document in a collection."""
    return {str(doc["_id"]): oid_to_uuid(str(doc["_id"])) for doc in docs}


def to_ts(val) -> str | None:
    """MongoDB datetime or ISO string → PostgreSQL timestamptz string."""
    if val is None:
        return None
    if isinstance(val, datetime):
        if val.tzinfo is None:
            val = val.replace(tzinfo=timezone.utc)
        return val.isoformat()
    if isinstance(val, str):
        return val
    return None


def to_date(val) -> str | None:
    """MongoDB datetime → PostgreSQL date string (YYYY-MM-DD)."""
    if val is None:
        return None
    if isinstance(val, datetime):
        return val.date().isoformat()
    if isinstance(val, str):
        return val[:10]
    return None


def normalize_bcrypt_prefix(hash_str: str) -> str:
    """GoTrue (Go) only handles $2a$ and $2b$ — normalize $2y$ from PHP/other systems."""
    if hash_str and hash_str.startswith("$2y$"):
        return "$2b$" + hash_str[4:]
    return hash_str


def resolve_fk(id_map: dict, oid) -> str | None:
    """Resolve a MongoDB ObjectId reference to a UUID. Returns None if oid is None or missing."""
    if oid is None:
        return None
    return id_map.get(str(oid))
```

```python
def deduplicate_payslips(rows: list) -> list:
    """
    Supabase has UNIQUE (employee_id, period_month, period_year).
    MongoDB has no such constraint. Keep last record per unique key.
    """
    seen = {}
    for row in rows:
        key = (row["employee_id"], row["period_month"], row["period_year"])
        seen[key] = row  # last write wins
    return list(seen.values())
```

```python

migrate.py — Radiance ERP MongoDB → Supabase ETL orchestrator.

Usage:
  cd migration/
  cp .env.migration.example .env.migration
  # Edit .env.migration with real credentials
  python migrate.py

Output:
  migration/output/01_profiles.sql
  migration/output/02_companies.sql
  ... (one file per table, named with FK order prefix)
  migration/id_map.json  (ObjectId → UUID mapping for reference)
"""

import os
import json
import psycopg2
from dotenv import load_dotenv
from psycopg2.extras import execute_values

load_dotenv(".env.migration")

from migration.extract import extract_all, COLLECTIONS
from migration.utils import build_id_map, resolve_fk
from migration.transform import (
    transform_profile, transform_company, transform_contact,
    transform_pipeline_stage, transform_deal, transform_task,
    transform_note, transform_invoice, transform_invoice_item,

OUTPUT_DIR = "migration/output"


def write_sql_file(filename: str, table: str, rows: list, columns: list):
    """Write rows as a single INSERT ... VALUES statement to a .sql file."""
    if not rows:
        print(f"  [SKIP] {table} — 0 rows")
        return

    os.makedirs(OUTPUT_DIR, exist_ok=True)
    path = os.path.join(OUTPUT_DIR, filename)

    def escape(val):
        if val is None:
            return "NULL"
        if isinstance(val, bool):
            return "TRUE" if val else "FALSE"
        if isinstance(val, (int, float)):
            return str(val)
        if isinstance(val, list):
            # PostgreSQL array literal: ARRAY['a','b']
            escaped_items = [f"'{str(v).replace(chr(39), chr(39)*2)}'" for v in val]
            return f"ARRAY[{', '.join(escaped_items)}]"
        if isinstance(val, dict):
            s = json.dumps(val, default=str).replace("'", "''")
            return f"'{s}'::jsonb"
        s = str(val).replace("'", "''")
        return f"'{s}'"

    col_list = ", ".join(columns)
    with open(path, "w") as f:
        f.write(f"-- Table: {table} ({len(rows)} rows)\n")
        f.write(f"INSERT INTO {table} ({col_list}) VALUES\n")
        value_lines = []
        for row in rows:
            vals = [escape(row.get(col)) for col in columns]
            value_lines.append(f"  ({', '.join(vals)})")
        f.write(",\n".join(value_lines))
        f.write("\nON CONFLICT DO NOTHING;\n")

    print(f"  [OK] {path} — {len(rows)} rows")


def main():
```
